اختبار واجهات Jetpack Compose

تعلم كيفية كتابة اختبارات فعالة لواجهات المستخدم الحديثة المبنية باستخدام Compose.

مثال: اختبار زر التفاعل

@get:Rule
val composeTestRule = createComposeRule()

@Test
fun myTest() {
    // إعداد الواجهة
    composeTestRule.setContent {
        MyButton(text = "انقر هنا")
    }

    // التحقق من وجود النص
    composeTestRule.onNodeWithText("انقر هنا").assertExists()
    
    // محاكاة النقر
    composeTestRule.onNodeWithText("انقر هنا").performClick()
}
